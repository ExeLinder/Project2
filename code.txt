class Calc
{
	public static double Add(double x, double y)
	{
		return x+y;
	}
	
	public static double Mult(double x, double y)
	{
		return x*y;
	}
	public static double Div(double x, double y)
	{
		return x/y;
	}
	public static double Sub(double x, double y)
	{
		return x-y;
	}
}

void main()
{
	Calc.Add(2,7)
}
